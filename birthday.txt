        // Create enhanced fireworks effect
        function createFirework(x, y, type = 'burst') {
            const colors = ['#ff6b6b', '#feca57', '#48dbfb', '#ff9ff3', '#54a0ff', '#5f27cd', '#00d2d3', '#ff9f43'];
            
            if (type === 'mega') {
                // Create mega firework with trails
                for (let i = 0; i < 40; i++) {
                    const firework = document.createElement('div');
                    firework.className = 'firework-burst';
                    firework.style.left = x + 'px';
                    firework.style.top = y + 'px';
                    
                    const color = colors[Math.random() * colors.length | 0];
                    const angle = (i / 40) * Math.PI * 2;
                    const distance = Math.random() * 200 + 100;
                    
                    firework.style.background = `radial-gradient(circle, ${color}, transparent)`;
                    firework.style.transform = `translate(${Math.cos(angle) * distance}px, ${Math.sin(angle) * distance}px)`;
                    firework.style.animationDelay = Math.random() * 0.5 + 's';
                    
                    document.body.appendChild(firework);
                    
                    setTimeout(() => {
                        if (document.body.contains(firework)) {
                            document.body.removeChild(firework);
                        }
                    }, 3000);
                }
                
                // Add trailing sparks
                for (let i = 0; i < 15; i++) {
                    setTimeout(() => {
                        const trail = document.createElement('div');
                        trail.className = 'firework-trail';
                        trail.style.left = (x + Math.random() * 100 - 50) + 'px';
                        trail.style.top = y + 'px';
                        trail.style.background = `linear-gradient(to top, ${colors[Math.random() * colors.length | 0]}, transparent)`;
                        
                        document.body.appendChild(trail);
                        
                        setTimeout(() => {
                            if (document.body.contains(trail)) {
                                document.body.removeChild(trail);
                            }
                        }, 1500);
                    }, i * 100);
                }
            } else {
                // Regular firework burst
                for (let i = 0; i < 25; i++) {
                    const firework = document.createElement('div');
                    firework.className = 'fireworks';
                    firework.style.left = x + 'px';
                    firework.style.top = y + 'px';
                    firework.style.animationDelay = Math.random() * 0.5 + 's';
                    
                    const color = colors[Math.random() * colors.length | 0];
                    firework.style.background = `radial-gradient(circle, ${color}, transparent)`;
                    
                    document.body.appendChild(firework);
                    
                    setTimeout(() => {
                        if (document.body.contains(firework)) {
                            document.body.removeChild(firework);
                        }
                    }, 2000);
                }
            }
        }

        // Countdown timer
        function updateCountdown() {
            const targetDate = new Date('July 25, 2025 00:00:00').getTime();
            const now = new Date().getTime();
            const distance = targetDate - now;

            if (distance < 0) {
                // Birthday has arrived!
                document.getElementById('timerContainer').style.display = 'none';
                document.querySelector('.birthday-message').style.display = 'none';
                document.getElementById('celebrationStarted').style.display = 'block';
                
                // Create spectacular fireworks show
                const fireworksShow = () => {
                    // Multiple mega fireworks
                    for (let i = 0; i < 5; i++) {
                        setTimeout(() => {
                            const x = Math.random() * (window.innerWidth - 200) + 100;
                            const y = Math.random() * (window.innerHeight - 200) + 100;
                            createFirework(x, y, 'mega');
                        }, i * 300);
                    }
                    
                    // Regular fireworks scattered
                    for (let i = 0; i < 8; i++) {
                        setTimeout(() => {
                            const x = Math.random() * window.innerWidth;
                            const y = Math.random() * window.innerHeight;
                            createFirework(x, y, 'burst');
                        }, i * 150 + 1000);
                    }
                };
                
                // Start initial fireworks show
                fireworksShow();
                
                // Continue with regular fireworks every 2 seconds
                setInterval(() => {
                    // Random mega fireworks
                    if (Math.random() < 0.3) {
                        const x = Math.random() * (window.innerWidth - 200) + 100;
                        const y = Math.random() * (window.innerHeight - 200) + 100;
                        createFirework(x, y, 'mega');
                    }
                    
                    // Regular fireworks
                    const x = Math.random() * window.innerWidth;
                    const y = Math.random() * window.innerHeight;
                    createFirework(x, y, 'burst');
                }, 2000);
                
                // Extra spectacular show every 10 seconds
                setInterval(fireworksShow, 10000);
                
                return;
            }

            const days = Math.floor(distance / (1000 * 60 * 60 * 24));
            const hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
            const minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            const seconds = Math.floor((distance % (1000 * 60)) / 1000);

            document.getElementById('days').textContent = days.toString().padStart(2, '0');
            document.getElementById('hours').textContent = hours.toString().padStart(2, '0');
            document.getElementById('minutes').textContent = minutes.toString().padStart(2, '0');
            document.getElementById('seconds').textContent = seconds.toString().padStart(2, '0');
        }

        // Random fireworks during countdown
        function randomFireworks() {
            if (Math.random() < 0.1) { // 10% chance every interval
                const x = Math.random() * window.innerWidth;
                const y = Math.random() * window.innerHeight;
                createFirework(x, y);
            }
        }

        createParticles();
        updateCountdown();
        setInterval(updateCountdown, 1000);
